CREATE TABLE CAD_FORMAPGTO (
  ID_FORMA INTEGER NOT NULL,
  NM_FORMA VARCHAR(40) NULL,
  IN_ENTRADA CHAR(1) NULL,
  NR_INTERVALO INTEGER NULL,
  NR_PARCELAS INTEGER NULL,
  PRIMARY KEY(ID_FORMA)
);

CREATE TABLE CAD_GENERO (
  ID_GENERO INTEGER NOT NULL,
  NM_GENERO VARCHAR(40) NULL,
  VL_FILME NUMERIC(10,2) NULL,
  PRIMARY KEY(ID_GENERO)
);

CREATE TABLE CAD_CLIENTE (
  ID_CLIENTE INTEGER NOT NULL,
  NM_CLIENTE VARCHAR(60) NULL,
  IN_ATIVO CHAR(1) NULL,
  PRIMARY KEY(ID_CLIENTE)
);

CREATE TABLE CAD_FILME (
  ID_FILME INTEGER NOT NULL,
  ID_GENERO INTEGER NOT NULL,
  NM_FILME VARCHAR(40) NULL,
  PRIMARY KEY(ID_FILME)
);

CREATE TABLE MOV_LOCACAO (
  ID_LOCACAO INTEGER NOT NULL,
  ID_FORMA INTEGER NOT NULL,
  ID_CLIENTE INTEGER NOT NULL,
  DT_LOCACAO DATE NULL,
  VL_TOTAL NUMERIC(10,2) NULL,
  PRIMARY KEY(ID_LOCACAO)
);

CREATE TABLE CAD_FILME_MOV_LOCACAO (
  ID_FILME INTEGER NOT NULL,
  ID_LOCACAO INTEGER NOT NULL,
  VL_FILME NUMERIC(8,2) NULL,
  QT_FILME INTEGER NULL,
  PRIMARY KEY(ID_FILME, ID_LOCACAO)
);

CREATE TABLE CAD_PARCELAS (
  ID_PARCELA INTEGER NOT NULL,
  ID_LOCACAO INTEGER NOT NULL,
  VL_PARCELA NUMERIC(10,2) NULL,
  VL_PAGO NUMERIC(10,2) NULL,
  DT_PAGAMENTO DATE NULL,
  DT_PAGO DATE NULL,
  DT_TRANSACAO DATE NULL,
  PRIMARY KEY(ID_PARCELA, ID_LOCACAO));
